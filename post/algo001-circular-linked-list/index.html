#include <iostream>

using namespace std;

// 队列长度，可以自己修改。
const int N = 10;

// 队列
struct Queue {
    int q[N];
    int hh = 1, tt = 0;

    void push(int x) {
        q[( ++ tt) % N] = x;
        if (hh / N && tt / N) {
            hh -= (int)min(hh / N, tt / N) * N;
            tt -= (int)min(hh / N, tt / N) * N;
        }
    }

    void pop() {
        hh ++ ;
    }

    int front() {
        return q[hh % N];
    }

    int back() {
        return q[tt % N];
    }

    bool empty() {
        return hh > tt;
    }

    int size() {
        return tt - hh + 1;
    }
};

// 调试
int main() {
    Queue q;
    q.push(1);
    q.pop();
    q.push(1);
    q.push(2);
    cout << (q.empty() ? 'Y' : 'N') << endl;
    q.push(3);
    q.pop();
    q.pop();
    q.pop();
    q.push(1);
    q.push(2);
    q.push(3);
    cout << q.front() << ' ';
    cout << q.back() << endl;
    q.push(4);
    q.push(5);
    cout << q.size();
    q.push(6);
    q.pop();
    q.pop();
    q.pop();
    q.pop();
    q.pop();
    q.pop();
    return 0;
}
